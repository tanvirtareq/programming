 7  bool binary_searchf(vector<long long int>v,long long int key)
 8  {
 9      lwrb=0;
10      uprb=v.size()-1;
11      long long int mid;
12      while(lwrb<=uprb)
13      {
14          mid=(lwrb+uprb)/2;
15          if(v[mid]==key)
16          {
17              target=mid;
18              return true;
19          }
20          else if(v[mid]>key)
21          {
22              uprb=mid-1;
23          }
24          else lwrb=mid+1;
25      }
26  
27      swap(lwrb, uprb);
28  
29      return false;
30  
31  }
